// 求n以内的全部质数  O(n)
// https://www.luogu.com.cn/problem/P3383
const int N = 1e8 + 10;
int vis[N];
vi prime;
void get_prime(int n)
{
    for (int i = 2; i <= n; i++)
    {
        if (vis[i] == 0)
            prime.push_back(i);
        for (int j = 0; 1LL * i * prime[j] <= n; j++)
        {
            vis[i * prime[j]] = 1;
            if (i % prime[j] == 0)  // 每个数字只用自身最小的质因数筛
                break;
        }
    }
}

// 筛法求n以内的欧拉函数值
// https://www.acwing.com/problem/content/description/876/
const int N = 1e8 + 10;
int vis[N], phi[N];
vi prime;
void euler_phi(int n)
{
    phi[1] = 1;
    for (int i = 2; i <= n; i++)
    {
        if (vis[i] == 0)
        {
            prime.push_back(i);
            phi[i] = i - 1;
        }
        for (int j = 0; 1LL * i * prime[j] <= n; j++)
        {
            int m = i * prime[j];
            vis[m] = 1;
            if (i % prime[j] == 0) // 若i被p[j]整除，那么i包含了m的全部质因子
            {
                phi[m] = prime[j] * phi[i];
                break;
            }
            else        // 若i不能被p[j]整除，则i和p[j]是互质的
            {
                phi[m] = (prime[j] - 1) * phi[i];
            }
        }
    }
}

// 筛法求n以内的约数个数
// 约数个数定理：若n = p1^a1 * p2^a2 * ... * pk^ak，则约数个数为(a1 + 1) * (a2 + 1) * ... * (ak + 1)
const int N = 1e8 + 10;
int vis[N];
vi prime;
int a[N];   // a[i]记录i的最小质因子的次数
int d[N];   // d[i]记录i的约数个数
void get_d(int n)
{
    d[1] = 1;
    for (int i = 2; i <= n; i++)
    {
        if (!vis[i])
        {
            prime.push_back(i);
            a[i] = 1;
            d[i] = 2;
        }
        for (int j = 0; i * prime[j] <= n; j++)
        {
            int m = i * prime[j];
            vis[m] = 1;
            if (i % prime[j] == 0)
            {
                a[m] = a[i] + 1;
                d[m] = d[i] / a[m] * (a[m] + 1);
                break;
            }
            else
            {
                a[m] = 1;
                d[m] = d[i] * 2;
            }
        }
    }
}

// 筛法求约数和
// 约数和定理：若n = p1^a1 * p2^a2 * ... * pk^ak，则约数和为(p1^(a1 + 1) - 1) / (p1 - 1) * (p2^(a2 + 1) - 1) / (p2 - 1) * ... * (pk^(ak + 1) - 1) / (pk - 1)
const int N = 1e8 + 10;
int vis[N];
// g[i]表示i的最小质因子的1+p^1+...+p^k
// f[i]表示i的约数和
int g[N], f[N];
vi prime;

void get_f(int n)
{ // 筛法求约数和
    g[1] = f[1] = 1;
    for (int i = 2; i <= n; i++)
    {
        if (!vis[i])
        {
            prime.push_back(i);
            g[i] = f[i] = i + 1;
        }
        for (int j = 0; i * prime[j] <= n; j++)
        {
            int m = i * prime[j];
            vis[m] = 1;
            if (i % prime[j] == 0)
            {
                g[m] = g[i] * prime[j] + 1;
                f[m] = f[i] / g[i] * g[m];
                break;
            }
            else
            {
                g[m] = prime[j] + 1;
                f[m] = f[i] * g[m];
            }
        }
    }
}

// 筛法求莫比乌斯函数
// 唯一分解定理：n = p1^a1 * p2^a2 * ... * pk^ak
// 莫比乌斯函数定义:
/*
 * μ(n) = 1, n = 1
 * μ(n) = 0, n有相同质因子
 * μ(n) = (-1)^k, k为n的不同质因子个数
 */
const int N = 1e8 + 10;
int vis[N];
int mu[N];
vi prime;

void get_mu(int n)
{
    mu[1] = 1;
    for (int i = 2; i <= n; i++)
    {
        if (!vis[i])
        {
            prime.push_back(i);
            mu[i] = -1;
        }
        for (int j = 0; i * prime[j] <= n; j++)
        {
            int m = i * prime[j];
            vis[m] = 1;
            if (i % prime[j] == 0)
            {
                mu[m] = 0;
                break;
            }
            else
                mu[m] = -mu[i];
        }
    }
}
