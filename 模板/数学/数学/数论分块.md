/*
 * 节目最初播出时，会有 m 名观众观看。若第 i 集投放有广告，记此时还剩有 c 名观众观看，则会产生c·pi 的收益；
 * 但播出后则会让观众的人数变为 c′ = ⌊c/di⌋，即第 i + 1 集只会剩有 c′ 名观众观看。如果在第 i 集没有投放广告，则不会产生收益，观众人数也不会变化。
 */
void solve()
{
    int n, m;
    cin >> n >> m;
    vi p(n), d(n);
    for (int i = 0; i < n; i++)
        cin >> p[i];
    for (int i = 0; i < n; i++)
        cin >> d[i];
    int res = 0;
    vi f(m + 1, 0);
    set<int> a;
    // m可能变成哪些值，最多根号n种
    for (int i = 1; i <= m; i++)
        a.insert(m / i);
    for (int i = 0; i < n; i++)
    {
        for (auto x : a)
        {
            f[x / d[i]] = max(f[x / d[i]], f[x] + x * p[i]);
        }
    }
    print(*max_element(all(f)));
}
