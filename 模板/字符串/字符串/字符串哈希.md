# 字符串哈希
```c++

const int MOD = 1e9 + 7;
const int BASE = 2333;
class Hash
{
public:
    vector<int> h, rh, p;
    Hash(const string &s)
    {
        int n = s.size();
        h.resize(n + 2, 0);
        rh.resize(n + 3, 0);
        p.resize(n + 2, 1);
        for (int i = 1; i <= n; i++)
        {
            p[i] = p[i - 1] * BASE % MOD;
            h[i] = (h[i - 1] * BASE + (s[i - 1] - 'a')) % MOD;
        }
        for (int i = n; i >= 1; i--)
        {
            rh[i] = (rh[i + 1] * BASE + (s[i - 1] - 'a')) % MOD;
        }
    }
    int get(int l, int r)
    {
        int x = h[r] - h[l - 1] * p[r - l + 1] % MOD;
        return (x + MOD) % MOD;
    }
    int getRev(int l, int r)
    {
        int x = rh[l] - rh[r + 1] * p[r - l + 1] % MOD;
        return (x + MOD) % MOD;
    }

    int concatHash(int h1, int len2, int h2)
    {
        return (h1 * p[len2] + h2) % MOD;
    }
};

void solve()
{
    string s;
    cin >> s;
    int n = s.size();
    int l = 0, r = n - 1;
    while (l <= r && s[l] == s[r])
        l++, r--;
    if (l > r)
        return print("Yes");
    string t = s.substr(l, r - l + 1);
    Hash H(t);
    int m = t.size();
    for (int i = 1; i < m; i++)
    {
        int t1 = H.getRev(1, i);
        int t2 = H.get(i + 1, m);
        int a = H.concatHash(t1, m - i, t2); // 正向哈希，左边补长度再加右边

        int d1 = H.getRev(i + 1, m);
        int d2 = H.get(1, i);
        int b = H.concatHash(d1, i, d2); // 反向哈希

        if (a == b)
            return print("Yes");
    }
    for (int i = m; i > 1; i--)
    {
        int lenA = i - 1, lenB = m - i + 1;
        int t1 = H.get(1, lenA);
        int t2 = H.getRev(i, m);
        int a = H.concatHash(t1, lenB, t2);

        int d1 = H.get(i, m);
        int d2 = H.getRev(1, lenA);
        int b = H.concatHash(d1, lenA, d2);

        if (a == b)
            return print("Yes");
    }

    print("No");
}
```
