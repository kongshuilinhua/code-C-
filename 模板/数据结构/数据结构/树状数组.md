# 树状数组
```c++

class FenwickTree
{
private:
    vector<int> tree;
    int size;

public:
    FenwickTree(int n) : size(n)
    {
        tree.resize(n + 2, 0);
    }

    void update(int index, int delta)
    {
        while (index <= size)
        {
            tree[index] += delta;
            index += index & -index;
        }
    }

    int query(int index)
    {
        int res = 0;
        while (index > 0)
        {
            res += tree[index];
            index -= index & -index;
        }
        return res;
    }
};
```

# 查询区间元素种类数

```c++
// https://www.luogu.com.cn/problem/P1972
const int mod = 998244353;
const int inf = 1e18;
// const int mod = 1e9 + 7;
struct x
{
    int l, r, idx;
};

bool cmp(const x &a, const x &b)
{
    return a.r < b.r;
}
void solve()
{
    int n, q;
    cin >> n;
    vi a(n);
    for (auto &x : a)
        cin >> x;
    cin >> q;
    vector<x> qrs(q);

    for (int i = 0; i < q; i++)
    {
        cin >> qrs[i].l >> qrs[i].r;
        qrs[i].idx = i;
    }
    // 右端点排序
    sort(all(qrs), cmp);
    FenwickTree tree(n);
    unordered_map<int, int> last;
    vi res(q);
    int cur = 0;
    for (auto [l, r, idx] : qrs)
    {
        while (cur < r)
        {
            cur += 1;
            int x = a[cur - 1];
            // 左边的点如果存在，消除它的影响
            if (last.count(x))
            {
                tree.update(last[x], -1);
            }
            tree.update(cur, 1);
            last[x] = cur;
        }
        res[idx] = tree.query(r) - tree.query(l - 1);
    }
    for (auto x : res)
    {
        print(x);
    }
}

    cin >> m;
    vector<ai3> qrs;
    for (int i = 0; i < m; i++)
    {
        int l, r;
        cin >> l >> r;
        qrs.push_back({l, r, i});
    }
    auto cmp = [&](ai3 &a, ai3 &b)
    {
        return a[1] < b[1];
    };
    sort(all(qrs), cmp);
    vi res(m);

    unordered_map<int, int> last;
    FenwickTree tree(n + 1);
    int cur = 0;
    for (auto [l, r, idx] : qrs)
    {
        while (cur < r)
        {
            cur++;
            int x = a[cur];
            if (last.count(x))
            {
                tree.update(last[x], x);
            }
            tree.update(cur, x);
            last[x] = cur;
        }
        res[idx] = tree.query(r) ^ tree.query(l - 1) ^ s1[r] ^ s1[l - 1];
    }
    for (auto x : res)
        print(x);
}
```
